---- for linux_ubuntu (example: user 'lab' on Ubuntu 24.04) ----
cd ~
mkdir -p repo/python-basics-for-netauto

cd repo/python-basics-for-netauto
git clone https://github.com/behrang-netauto/python-basics-for-netauto.git .

sudo apt update
sudo apt install -y python3 python3-venv python3-pip git

python3 --version  # should be 3.12.x on Ubuntu 24.04+

cd python-basics-for-netauto

python3 -m venv .venv
#ubuntu@ubuntu:~/repo/python-basics-for-netauto$ du -sh .venv/
#16M     .venv/
source .venv/bin/activate

pip install --upgrade pip
pip install -r requirements.txt
#(.venv) ubuntu@ubuntu:~/repo/python-basics-for-netauto$ du -sh .venv/
#1.8G    .venv/

---- for macOS (zsh/bash, with Homebrew) ----
